#pragma once
#include <Mudbox/mudbox.h>
class CgFile
{
public:
CgFile( const QString &sFileName )
{
QFile f( sFileName );
if ( f.open( QIODevice::ReadOnly ) )
{
m_aContent.SetItemCount( f.size()+1 );
f.read( m_aContent.baseAddr(), f.size() );
m_aContent[int(f.size())] = 0;
}
}
const mudbox::Store<char> &Buffer( void ) const
{
return m_aContent;
}
private:
mudbox::Store<char> m_aContent;
};
