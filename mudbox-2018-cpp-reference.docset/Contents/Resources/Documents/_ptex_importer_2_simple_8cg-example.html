PtexImporter/Simple.cg
struct VertIn {
float2 V : TEXCOORD0;
};
float4 main( VertIn In,
uniform sampler2D quadTex : TEXUNIT0,
uniform float2 V0,
uniform float2 V1,
uniform float2 V2,
uniform float2 V3
) : COLOR
{
float2 p1p2 = lerp(V0, V1, In.V.x);
float2 p4p3 = lerp(V3, V2, In.V.x);
float2 p = lerp(p1p2, p4p3, In.V.y);
return tex2D(quadTex, p);
}
