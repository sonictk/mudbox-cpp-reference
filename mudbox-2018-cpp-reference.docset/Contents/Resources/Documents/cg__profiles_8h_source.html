52 /*
53  * The following macro invocations define the supported CG profiles.
54  *
55  * The macros have the form :
56  *
57  * CG_PROFILE_MACRO(name, compiler_id, compiler_id_caps, compiler_opt, int_id, vertex_profile)
58  *
59  * name : The name of the profile. Used consistently with the API.
60  * compiler_id : The identifier string for the profile used by the compiler.
61  * compiler_id_caps : compiler_id in caps.
62  * compiler_opt : The command-line switch used to force compilation into
63  * the profile.
64  * int_id : Integer enumerant associated with this bind location.
65  * vertex_profile : Non-zero if this is a vertex profile, otherwise it
66  * is considered to be a fragment profile.
67  *
68  *
69  */
70

76 #endif
