110 #define __CS_LOOP(ptr, op_val, op_string, pre, post) __GNU_EXTENSION ({ \
111  volatile int old_val, new_val; \
112  __asm__ __volatile__(pre \
113  " l %0,0(%3)\n" \
114  "0: lr %1,%0\n" \
115  op_string " %1,%4\n" \
116  " cs %0,%1,0(%3)\n" \
117  " jl 0b\n" \
118  post \
119  : "=&d" (old_val), "=&d" (new_val), \
120  "=m" (*ptr) \
121  : "a" (ptr), "d" (op_val), \
122  "m" (*ptr) \
123  : "cc", "memory" ); \
124  new_val; \
125 })
